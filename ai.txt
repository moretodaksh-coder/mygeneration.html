<script>
// --- JS ---
const messages = document.getElementById('messages');
const userInput = document.getElementById('userInput');
const sendBtn = document.getElementById('sendBtn');
const aiOrb = document.getElementById('aiOrb');

const aiResponses = [
  "Hello! I am your talkative AI.",
  "Welcome to My Generation!",
  "I can chat with you and even speak!",
  "Isn't this visual effect cool?",
  "Type anything and I will respond."
];

function addMessage(text, sender) {
  const msg = document.createElement('div');
  msg.classList.add('message', sender);
  msg.textContent = text;
  messages.appendChild(msg);
  messages.scrollTop = messages.scrollHeight;
}

function aiSpeak(text) {
  addMessage(text, 'ai');

  const utterance = new SpeechSynthesisUtterance(text);
  utterance.lang = 'en-US';
  utterance.rate = 1;
  speechSynthesis.speak(utterance);

  aiOrb.style.animation = 'pulse 0.5s infinite';
  utterance.onend = () => {
    aiOrb.style.animation = 'pulse 2s infinite';
  }
}

function aiReply() {
  const reply = aiResponses[Math.floor(Math.random() * aiResponses.length)];
  aiSpeak(reply);
}

sendBtn.addEventListener('click', () => {
  const text = userInput.value.trim();
  if (!text) return;
  addMessage(text, 'user');
  userInput.value = '';
  setTimeout(aiReply, 500);
});

userInput.addEventListener('keypress', (e) => {
  if (e.key === 'Enter') sendBtn.click();
});
</script>
